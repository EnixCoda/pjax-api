<div class="col-md-4">
## Installation

```
$ npm i pjax-api
```

```html
<script src="/js/pjax-api.js"></script>
<script>
var Pjax = require('pjax-api').Pjax;
new Pjax({
  areas: [
    // try to use the first query.
    '#header, #primary',
    // fallback, retrying with the second query.
    '#container',
    // fallback.
    'body'
  ]
});
</script>
```
